#ifndef LINKEDLIST_H
#define LINKEDLIST_H
#include "stdbool.h"

typedef struct Node {
    // A node is an individual struct that makes up a linked list. If we were in Java, Linked List would extend Node
    // But since we're in C, I'm just going to use Node in linked list.
    struct Node* next;
    void* data;
}Node;

typedef struct LinkedList {
    Node* head;
    Node* tail;
    // Just wondering if we should track size
    int size;
}LinkedList;

void addAtTail(LinkedList* list, void* data);
void addAtHead(LinkedList* list, void* data);
void* removeAtTail(LinkedList* list);
void* removeAtHead(LinkedList* list);
void* peakHead(LinkedList* list);
void* peakTail(LinkedList* list);
void removeNodeAt(LinkedList* list, int index);
bool contains(LinkedList* list, void* elements, int (*cmp)(void *, void *));
bool isEmpty(LinkedList* stack);
LinkedList* initLinkedList ();

#endif //LINKEDLIST_H

#include <stdio.h>
#include <string.h>
#include <stdlib.h>//allows use of malloc and calloc
#include "Stack.h"

Stack* initStack() {
    Stack* stack = malloc(sizeof(Stack));
    stack->list = initLinkedList();  // Initialize the internal linked list
    return stack;
};

bool isStackEmpty(Stack* stack) {
    return isEmpty(stack->list);
}

void pop(Stack* stack) {
    if (isStackEmpty(stack)) {
        printf("Stack is empty\n");
        return;
    }
    removeAtHead(stack->list);
}

void* peekStack(Stack* stack) {
    if (isStackEmpty(stack)) {
        printf("Stack is empty\n");
        return NULL;
    }
    return stack->list->head->data;
}

void push(Stack* stack, void* data) {
    addAtHead(stack->list, data);
}

#ifndef MAZE_H
#define MAZE_H
#include <stdbool.h>
#include "LinkedList.h"

//structure to represent coordinates in the maze
typedef struct Coordinate {
    int y;
    int x;
} Coordinate;

typedef struct Maze {
    char** mazeArray;
    int rows;
    int cols;
    LinkedList* visited;
    Coordinate** spaces;
} Maze;


// Reads the maze from a file and returns a 2D array of characters
Maze* readFile(const char* fileName);

// Finds the starting coordinate in the maze, returns (-1,-1) if the start is not found
Coordinate findStart(Maze *maze);

// Finds valid neighbors around a given coordinate and returns them as a linked list
//Order of the linked list will be up, down, left, right
LinkedList* findValidNeighbors(Maze *maze, Coordinate z);

// Adds a coordinate to the global visited list
void addVisited(Maze *maze, Coordinate z);
int coordinateCmp(void* a, void *b);


// Checks if a coordinate is the end of the maze
bool isEnd(Maze *maze, Coordinate z);

// Prints the maze to the console
void printMaze(Maze *maze);

Maze* genMaze(int width, int height);

bool solvable(Maze *maze);

#endif //MAZE_H
